# Installation instructions:
#
# 1. Basic installation (CPU only):
#    pip install -r requirements.txt
#
# 2. For better performance, build dlib with optimizations:
#    # Install build dependencies
#    sudo apt-get install cmake build-essential
#
#    # Build dlib from source with optimizations
#    git clone https://github.com/davisking/dlib.git
#    cd dlib
#    mkdir build && cd build
#    cmake .. -DUSE_AVX_INSTRUCTIONS=ON -DUSE_SSE4_INSTRUCTIONS=ON
#    cmake --build . --config Release
#    cd ..
#    python setup.py install
#
# 3. For GPU acceleration (NVIDIA GPU with CUDA):
#    # Install CUDA toolkit first (10.0+)
#    # Then build dlib with CUDA support
#    cmake .. -DDLIB_USE_CUDA=ON -DUSE_AVX_INSTRUCTIONS=ON
#    cmake --build . --config Release
#    python setup.py install
#
# 4. Download pre-trained models:
#    bash scripts/download_models.sh
#
# 5. Test installation:
